### Number to Pattern 5
def getVal(k):
    if k==0:
        return 'A'
    if k==1:
        return 'C'
    if k==2:
        return 'G'
    return 'T'
def NUMBERTOPATTERN(index, k):
    initial=index
    str=""
    while (k>0):
        rem=int(initial%4)
        str=str+getVal(rem)
        initial=initial/4
        k=k-1
    str=str[::-1]
    return str
print(NUMBERTOPATTERN(9904,7))
